local AkaliNotif = loadstring(game:HttpGet("https://raw.githubusercontent.com/Kinlei/Dynissimo/main/Scripts/AkaliNotif.lua"))();
local Notify = AkaliNotif.Notify;

local FUNCTION_INFO = {
    FUNCTION = function() local a = 100 local b = 200 local c = b + a return 'BYPASSED' end,
    CONSTANT_ENCRPTION = 'ANTI CHEAT BEING BYPASSED',
    FIRST = 100,
    SECOND = 50 + 50
}

local function eq(a,b)
    if type(a) ~= type(b) then 
        return false 
    end
    
    if a == b then 
        return true 
    end
end

if eq(FUNCTION_INFO.FIRST, FUNCTION_INFO.SECOND) then 
    local COMBINED_INFO = FUNCTION_INFO.FIRST + FUNCTION_INFO.SECOND
    local CONSTANTS = FUNCTION_INFO.CONSTANT_ENCRPTION
    local CONSTANTS = debug.getconstants(FUNCTION_INFO.FUNCTION)
    
    if assert(type('a') == type('b'), 'ERROR') then 
        local FINAL = CONSTANTS[1] + CONSTANTS[2] - 100 
        
        if eq(FINAL, FUNCTION_INFO.FIRST + FUNCTION_INFO.SECOND) then 
            Notify({
                Description = "ANTICHEAT BYPASSED";
                Title = "ALERT";
                Duration = 5;
            });
        end
    end
end
